Usage Examples

Example #1
<selenium baseurl="http://somewhere.com">
      <chooseOkOnNextConfirmation/>
      <runScript script="window.confirm('Are you sure?')"/>
      <assert>
        <isConfirmationPresent/>
    </assert>
      <getConfirmation property="confirm.msg"/>
      <assert>
        <eq op1="${confirm.msg}" op2="Are you sure?"/>
    </assert>
  </selenium>
